    import fastplotlib as fpl
    import imageio.v3 as iio  # not a fastplotlib dependency, only used for examples

    a = iio.imread("imageio:camera.png")
    iw = fpl.ImageWidget(data=a, cmap="viridis", figure_kwargs={"size": (700, 560)})
